      def softmax(x):
          '''
          softmax
          
          
              x:
          
          '''
          exp_x = np.exp(x-np.max(x))
          return exp_x/exp_x.sum(axis=0)
      def rnn_cell_forward(xt,a_prev,parameters):
          '''
          RNN
          
              xttnumpy.array,:(n_x,m)
              a_prev: t-1numpy.array:(n_a,m)
              paremeters:Python
                  Wax:numpy.array,(n_a,n_x)
                  Waa:numpy.array,n_a,n_a
                  Wya:numpy.array,(n_y,n_a)
                  ba:numpy.array(n_a,1)
                  by:numpy.array(n_y,1)
          
              a_next::(n_a,m)
              yt_pred:tn_y,m
              cache:a_next,a_prev,xt,parameters    
          '''
          
          #parameters
          Wax = parameters["Wax"]
          Waa = parameters["Waa"]
          Wya = parameters["Wya"]
          ba = parameters["ba"]
          by = parameters["by"]
          
          #
          a_next = np.tanh(np.dot(Wax,xt)+np.dot(Waa,a_prev)+ba)
          #
          yt_pred = softmax(np.dot(Wya,a_next)+by)
          #
          cache = (a_next,a_prev,xt,parameters)
          
          return a_next,yt_pred,cache

      def rnn_forward(x,a0,parameters):
          '''
          RNN
          
              xtime_step,(n_x,m,T_x)
              a0:,(n_a,m)
              parameters:python,
                  Wax:numpy.array,(n_a,n_x)
                  Waa:numpy.array,n_a,n_a
                  Wya:numpy.array,(n_y,n_a)
                  ba:numpy.array(n_a,1)
                  by:numpy.array(n_y,1)
          
              a:time_step,(n_a,m,T_x)
              y_pred:time_step(n_y,m,T_x)
              caches:(list of caches,x)
          '''
          #caches
          caches = []
          
          #xWya
          n_x,m,T_x = x.shape
          n_y,n_a = parameters["Wya"].shape
          
          #ay
          a = np.zeros([n_a,m,T_x])
          y_pred = np.zeros([n_y,m,T_x])
          
          #a_next
          a_next = a0
          
          #time_steps
          for t in range(T_x):
              #rnn_cell_forwarda_next,yt_pred,cache
              a_next,yt_pred,cache = rnn_cell_forward(x[:,:,t],a_next,parameters)
              #a_nexta
              a[:,:,t] = a_next
              #yt_predy
              y_pred[:,:,t] = yt_pred
              #cachecaches
              caches.append(cache)
          
          #caches
          caches = (caches,x)
          
          return a,y_pred,caches

      def sigmoid(x):
          return 1/(1+np.exp(-x))
      def lstm_cell_forward(xt,a_prev,c_prev,parameters):
          '''
          lstm
          
              xttnumpy.array,:(n_x,m)
              a_prev: t-1numpy.array:(n_a,m)
              c_prev: t-1numpy.array,n_a,m
              paremeters:Python
                  Wf:numpy.array(n_a,n_a+n_x)
                  bf:numpy.array(n_a,1)
                  Wi:numpy.array,(n_a,n_x+n_a)
                  Wi:numpy.array,(n_a,1)
                  Wc:tanhnumpy.array,(n_a,n_a+n_x)
                  bc:tanhnumpy.array,(n_a,n_a+n_x)
                  Wo:numpy.array,(n_a,n_a+n_x)
                  bo:numpy.array,(n_a,1)
                  Wy:numpy.array,(n_y,n_a)
                  by:numpy.array,(n_y,n_a)
          
              a_next:(n_a,m)
              c_next:,(n_a,m)
              yt_pred:t,(n_y,m)
              cache:,(a_next,c_next,a_prev,c_prev,xt,parameters)
          '''
          
          #parameters
          Wf = parameters["Wf"]
          bf = parameters["bf"]
          Wi = parameters["Wi"]
          bi = parameters["bi"]
          Wc = parameters["Wc"]
          bc = parameters["bc"]
          Wo = parameters["Wo"]
          bo = parameters["bo"]
          Wy = parameters["Wy"]
          by = parameters["by"]
          
          #xtWy
          n_x,m = xt.shape
          n_y,n_a = Wy.shape
          
          #a_prevxt
          concat= np.zeros([n_a+n_x,m])
          concat[:n_a,:] = a_prev
          concat[n_a:,:] = xt
          #concat = np.concatenate(a_prev,xt)
          
          #
          ft = sigmoid(np.dot(Wf,concat)+bf)
          #
          it = sigmoid(np.dot(Wi,concat)+bi)
          cct = np.tanh(np.dot(Wc,concat)+bc)
          #
          c_next = ft*c_prev+it*cct
          ot = sigmoid(np.dot(Wo,concat)+bo)
          a_next = ot*np.tanh(c_next)
          
          #LSTM
          yt_pred = softmax(np.dot(Wy,a_next)+by)
          
          #cache
          cache = (a_next,c_next,a_prev,c_prev,ft,it,cct,ot,xt,parameters)
          
          return a_next,c_next,yt_pred,cache

      def rnn_cell_backward(da_next,cache):
          '''
          rnn
          
              da_next:
              cache:python
          
              gradients:python
                  dx:,(n_x,m)
                  da_prev:(n_a,m)
                  dWax:,(n_a,n_x)
                  dWaa:,(n_a,n_a)
                  dba:,(n_a,1)
          '''
          #cache
          (a_next,a_prev,xt,parameters) = cache
          
          #parameters
          Wax = parameters["Wax"]
          Waa = parameters["Waa"]
          Wya = parameters["Wya"]
          ba = parameters["ba"]
          by = parameters["by"]
          
          #tanha_next
          dtanh = (1-a_next*a_next)*da_next
          
          #Wax
          dxt = np.dot(Wax.T,dtanh)
          dWax = np.dot(dtanh,xt.T)
          
          #Waa
          da_prev = np.dot(Waa.T,dtanh)
          dWaa = np.dot(dtanh,da_prev.T)
          
          #b
          dba = np.sum(dtanh,keepdims=True,axis=-1)
          
          #
          gradients = {"dxt":dxt,"da_prev":da_prev,"dWax":dWax,"dWaa":dWaa,"dba":dba}
          
          return gradients

      def rnn_backward(da,caches):
          '''
          RNN
          
              da:(n_a,m,T_x)
              caches:
          
              gradients:python
                  dx:,(n_x,m,T_x)
                  da0:(n_a,m)
                  dWax:,(n_a,n_x)
                  dWaa:,(n_a,n_a)
                  dba:,(n_a,1)
          '''
          #caches
          (caches,x) = caches
          (a1,a0,x1,parameters) = caches[0]
          
          #dax1
          n_a,m,T_x = da.shape
          n_x,m = x1.shape
          
          #
          dx = np.zeros([n_x,m,T_x])
          dWax = np.zeros([n_a,n_x])
          dWaa = np.zeros([n_a,n_a])
          dba = np.zeros([n_a,1])
          da0 = np.zeros([n_a,m])
          da_prevt = np.zeros([n_a,m])
          
          #
          for t in reversed(range(T_x)):
              #rnn_cell_backward
              gradients = rnn_cell_backward(da[:,:,t]+da_prevt,caches[t])
              dxt = gradients["dxt"]
              da_prevt = gradients["da_prev"]
              dWaxt = gradients["dWax"]
              dWaat = gradients["dWaa"]
              dbat = gradients["dba"]
              
              dx[:,:,t] = dxt
              dWax += dWaxt
              dWaa += dWaat
              dba += dbat
          
          da0 = da_prevt
          
          #
          gradients = {"dx":dx,"da0":da0,"dWax":dWax,"dWaa":dWaa,"dba":dba}
          
          return gradients
